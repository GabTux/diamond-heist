def initialize(object_pool, source_x, source_y, direction)
  super(object_pool)
  @x = source_x
  @y = source_y
  @direction = direction
  BulletPhysics.new(self, object_pool)
  BulletGraphics.new(self)
  BulletSounds.play
end
